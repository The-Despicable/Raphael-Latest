```python
# NEW FILE: orchestrator/skills_bridge.py
import yaml
import glob
import os
from pathlib import Path

SKILLS_BASE = "/tmp/Anthropic-Cybersecurity-Skills/skills"

class SkillsBridge:
    def __init__(self):
        self.skill_index = self._build_index()
    
    def _build_index(self):
        """Catalog all 817 skills with their metadata and executability"""
        index = {}
        for subdomain in Path(SKILLS_BASE).iterdir():
            if subdomain.is_dir():
                index[subdomain.name] = []
                for skill_file in subdomain.glob("*.md"):
                    skill_meta = self._parse_skill(skill_file)
                    index[subdomain.name].append(skill_meta)
        return index
    
    def _parse_skill(self, path):
        """Extract YAML frontmatter and executable paths"""
        with open(path) as f:
            content = f.read()
        # Parse frontmatter
        if content.startswith('---'):
            _, frontmatter, body = content.split('---', 2)
            meta = yaml.safe_load(frontmatter)
        else:
            meta = {}
        
        # Check for executable scripts
        skill_dir = path.parent
        scripts_dir = skill_dir / "scripts"
        executables = list(scripts_dir.glob("*")) if scripts_dir.exists() else []
        
        return {
            "name": meta.get("name", path.stem),
            "description": meta.get("description", ""),
            "mitre_attack": meta.get("mitre_attack", []),
            "nist_csf": meta.get("nist_csf", []),
            "path": str(path),
            "has_executable": len(executables) > 0,
            "executables": [str(e) for e in executables]
        }
    
    def get_skills_for_pipeline(self, pipeline_name):
        """Return relevant skills for a given Raphael pipeline"""
        mapping = {
            "scanners": ["cloud-security", "network-security", "threat-intelligence"],
            "test": ["web-application-security", "vulnerability-management", "penetration-testing"],
            "postex": ["red-teaming", "identity-access-management", "malware-analysis"],
            "phishing": ["red-teaming", "identity-access-management"],
            "exfil": ["cloud-security", "network-security"],
            "anti_forensics": ["digital-forensics", "incident-response"]
        }
        subdomain = mapping.get(pipeline_name, [])
        return {s: self.skill_index.get(s, []) for s in subdomain}
```

```python
                return self._legacy_sqli_test(target)
```
